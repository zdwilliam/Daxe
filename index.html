    <script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script>

    <!-- 引入 OrbitControls 用于鼠标控制 -->

    <script src="https://cdn.jsdelivr.net/npm/three@0.128.0/examples/js/controls/OrbitControls.js"></script>

 

    <script>

        // --- 1. 初始化场景 ---

        const container = document.getElementById('canvas-container');

        const scene = new THREE.Scene();

        scene.background = new THREE.Color(0x1a1a1a); // 深灰色背景

        // 添加一点环境雾气

        scene.fog = new THREE.FogExp2(0x1a1a1a, 0.02);

 

        const camera = new THREE.PerspectiveCamera(45, window.innerWidth / window.innerHeight, 0.1, 1000);

        camera.position.set(3, 2, 5); // 初始相机位置

 

        // 渲染器设置

        const renderer = new THREE.WebGLRenderer({ antialias: true, preserveDrawingBuffer: true });

        renderer.setSize(window.innerWidth, window.innerHeight);

        renderer.shadowMap.enabled = true;

        renderer.toneMapping = THREE.ACESFilmicToneMapping;

        container.appendChild(renderer.domElement);

 

        // 鼠标控制

        const controls = new THREE.OrbitControls(camera, renderer.domElement);

        controls.enableDamping = true;

        controls.dampingFactor = 0.05;

        controls.maxPolarAngle = Math.PI / 2 - 0.05; // 禁止看到地底下

 

        // --- 2. 灯光系统 ---

        const ambientLight = new THREE.AmbientLight(0xffffff, 0.6);

        scene.add(ambientLight);

 

        const dirLight = new THREE.DirectionalLight(0xffffff, 1);

        dirLight.position.set(5, 10, 7);

        dirLight.castShadow = true;

        scene.add(dirLight);

 

        const rimLight = new THREE.SpotLight(0x00aaff, 2); // 蓝色轮廓光，增加赛博感

        rimLight.position.set(-5, 2, -5);

        rimLight.lookAt(0,0,0);

        scene.add(rimLight);

 

        // --- 3. 创建“概念”Dax e 模型 (由于没有真实模型文件，我们用代码拼一个近似形状) ---

        // 真实项目中，这里应该使用 GLTFLoader 加载 .glb 文件

       

        const bikeGroup = new THREE.Group();

        scene.add(bikeGroup);

 

        // 材质定义

        const paintMaterial = new THREE.MeshStandardMaterial({

            color: 0xB85C38, roughness: 0.2, metalness: 0.6, envMapIntensity: 1

        }); // 车漆

        const chromeMaterial = new THREE.MeshStandardMaterial({

            color: 0xffffff, roughness: 0.0, metalness: 1.0

        }); // 镀铬

        const rubberMaterial = new THREE.MeshStandardMaterial({

            color: 0x111111, roughness: 0.9

        }); // 轮胎

        const seatMaterial = new THREE.MeshStandardMaterial({

            color: 0x111111, roughness: 0.8

        }); // 座椅

 

        // A. T型车架 (模拟 Dax 经典 T-bone)

        const frameGeo = new THREE.CylinderGeometry(0.25, 0.3, 2.5, 32);

        const frame = new THREE.Mesh(frameGeo, paintMaterial);

        frame.rotation.z = Math.PI / 2;

        frame.position.y = 0.8;

        frame.castShadow = true;

        frame.name = "frame";

        bikeGroup.add(frame);

 

        // B. 轮子

        function createWheel(zPos) {

            const wheelGroup = new THREE.Group();

            // 轮胎

            const tireGeo = new THREE.TorusGeometry(0.5, 0.15, 16, 100);

            const tire = new THREE.Mesh(tireGeo, rubberMaterial);

            tire.name = "tire_mesh";

            tire.castShadow = true;

            // 轮毂

            const rimGeo = new THREE.CylinderGeometry(0.35, 0.35, 0.1, 32);

            const rim = new THREE.Mesh(rimGeo, chromeMaterial);

            rim.rotation.x = Math.PI / 2;

           

            wheelGroup.add(tire);

            wheelGroup.add(rim);

            wheelGroup.position.set(zPos, 0.5, 0);

            return wheelGroup;

        }

        const frontWheel = createWheel(1.3);

        const rearWheel = createWheel(-1.3);

        bikeGroup.add(frontWheel);

        bikeGroup.add(rearWheel);

       

        // 前叉

        const forkGeo = new THREE.CylinderGeometry(0.05, 0.05, 1.2);

        const fork = new THREE.Mesh(forkGeo, chromeMaterial);

        fork.position.set(1.1, 1.0, 0);

        fork.rotation.z = -0.3;

        bikeGroup.add(fork);

 

        // C. 座椅

        const seatGeo = new THREE.BoxGeometry(1.2, 0.15, 0.4);

        const seat = new THREE.Mesh(seatGeo, seatMaterial);

        seat.position.set(-0.2, 1.05, 0);

        seat.name = "seat";

        bikeGroup.add(seat);

 

        // D. 地面反射

        const gridHelper = new THREE.GridHelper(20, 20, 0x444444, 0x222222);

        scene.add(gridHelper);

 

 

        // --- 4. 可选改装件 (默认隐藏) ---

 

        // 改装件：复古圆灯

        const headLightGroup = new THREE.Group();

        const lightCase = new THREE.Mesh(new THREE.SphereGeometry(0.15, 32, 32), chromeMaterial);

        const lightGlass = new THREE.Mesh(new THREE.CircleGeometry(0.13, 32), new THREE.MeshStandardMaterial({color: 0xffffee, emissive: 0xffffee, emissiveIntensity: 0.5}));

        lightGlass.position.z = 0.12;

        lightGlass.rotation.y = Math.PI/2; // 朝前

        lightCase.add(lightGlass);

        headLightGroup.position.set(1.4, 1.2, 0);

        headLightGroup.visible = false; // 默认关闭

        bikeGroup.add(headLightGroup);

 

        // 改装件：中置货架

        const rackGeo = new THREE.BoxGeometry(0.8, 0.02, 0.2);

        const rack = new THREE.Mesh(rackGeo, new THREE.MeshStandardMaterial({color: 0x111111}));

        rack.position.set(0.2, 0.98, 0); // 在车架上方

        rack.visible = false;

        bikeGroup.add(rack);

 

        // --- 5. 交互逻辑函数 ---

 

        // 更改车漆颜色

        window.changeBodyColor = function(colorHex) {

            const frame = bikeGroup.getObjectByName('frame');

            frame.material.color.setHex(colorHex);

            // 材质光泽度微调

            if(colorHex === 0x111111) { // 哑光黑特殊处理

                frame.material.roughness = 0.7;

                frame.material.metalness = 0.1;

            } else {

                frame.material.roughness = 0.2;

                frame.material.metalness = 0.6;

            }

        };

 

        // 更改座椅颜色

        window.changeSeatColor = function(colorHex) {

            const seat = bikeGroup.getObjectByName('seat');

            seat.material.color.setHex(colorHex);

        };

 

        // 切换零件显示

        window.togglePart = function(partName, isChecked) {

            if(partName === 'headlight') headLightGroup.visible = isChecked;

            if(partName === 'rack') rack.visible = isChecked;

        };

 

        // 切换轮胎样式 (通过改变粗细模拟越野胎)

        window.toggleTires = function(isOffroad) {

            bikeGroup.children.forEach(child => {

                if(child.children) {

                    child.children.forEach(part => {

                        if(part.name === 'tire_mesh') {

                            if(isOffroad) {

                                part.scale.set(1, 1, 1.5); // 变宽

                                part.material.roughness = 1.0;

                            } else {

                                part.scale.set(1, 1, 1);

                                part.material.roughness = 0.9;

                            }

                        }

                    })

                }

            });

        };

 

        // 灵感库一键方案

        window.applyPreset = function(styleName) {

            if(styleName === 'classic') {

                changeBodyColor(0xD32F2F); // 红

                changeSeatColor(0x111111); // 黑座

                document.getElementById('mod-headlight').checked = true; togglePart('headlight', true);

                document.getElementById('mod-rack').checked = true; togglePart('rack', true);

                document.getElementById('mod-tires').checked = false; toggleTires(false);

            }

            else if(styleName === 'tech') {

                changeBodyColor(0x1976D2); // 蓝

                changeSeatColor(0xEEEEEE); // 白座

                document.getElementById('mod-headlight').checked = false; togglePart('headlight', false);

                document.getElementById('mod-rack').checked = false; togglePart('rack', false);

            }

            else if(styleName === 'scrambler') {

                changeBodyColor(0x4CAF50); // 绿

                changeSeatColor(0x5D4037); // 棕座

                document.getElementById('mod-headlight').checked = true; togglePart('headlight', true);

                document.getElementById('mod-tires').checked = true; toggleTires(true);

            }

        }

 

        // 拍照保存

        window.saveSnapshot = function() {

            renderer.render(scene, camera);

            const dataURL = renderer.domElement.toDataURL('image/png');

            const link = document.createElement('a');

            link.download = 'My-Dax-Custom.png';

            link.href = dataURL;

            link.click();

        };

 

        // --- 6. 动画循环 ---

        function animate() {

            requestAnimationFrame(animate);

            controls.update();

            renderer.render(scene, camera);

        }

       

        // 窗口大小自适应

        window.addEventListener('resize', () => {

            camera.aspect = window.innerWidth / window.innerHeight;

            camera.updateProjectionMatrix();

            renderer.setSize(window.innerWidth, window.innerHeight);

        });

 

        animate();

 

    </script>
